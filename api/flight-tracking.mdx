## Search flights

<CodeGroup>
```bash cURL
curl -X GET "https://api.aero.com/v1/flight/search?iata=UA123&icao=UAL123&date=2024-03-20&timezone=America/Los_Angeles" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.aero.com/v1/flight/search?iata=UA123&icao=UAL123&date=2024-03-20&timezone=America/Los_Angeles', {
  headers: {
    'Authorization': 'Bearer YOUR_JWT_TOKEN'
  }
});
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.aero.com/v1/flight/search',
    params={
        'iata': 'UA123',
        'icao': 'UAL123',
        'date': '2024-03-20',
        'timezone': 'America/Los_Angeles'
    },
    headers={'Authorization': 'Bearer YOUR_JWT_TOKEN'}
)
data = response.json()
```
</CodeGroup>

<ParamField path="GET" type="/v1/flight/search">
  Search for flights by identifier and optional date. Returns lightweight flight summaries for selection.
</ParamField>

### Query parameters

<ParamField query="iata" type="string" required>
  IATA flight number
  
  **Example:** `UA123`
</ParamField>

<ParamField query="icao" type="string" required>
  ICAO flight number
  
  **Example:** `UAL123`
</ParamField>

<ParamField query="date" type="string">
  Flight date (ISO format). Filters flights by date if provided
  
  **Example:** `2024-03-20`
</ParamField>

<ParamField query="timezone" type="string" required>
  The timezone from which the request is being made
  
  **Example:** `America/Los_Angeles`
</ParamField>

### Response

<ResponseField name="flights" type="array">
  Array of flight candidates
  
  <Expandable title="Flight candidate object">

      FlightAware flight ID for fetching full details
    </ResponseField>
    
    <ResponseField name="ident" type="string">
      Flight identifier (e.g., "AC113")
    </ResponseField>
    
    <ResponseField name="origin" type="object">
      Origin airport information
      
      <Expandable title="properties">
        <ResponseField name="code" type="string">
          Airport code
        </ResponseField>
        
        <ResponseField name="codeIata" type="string">
          IATA code
        </ResponseField>
        
        <ResponseField name="codeIcao" type="string">
          ICAO code
        </ResponseField>
        
        <ResponseField name="name" type="string">
          Airport name
        </ResponseField>
        
        <ResponseField name="city" type="string">
          City name
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="destination" type="object">
      Destination airport information (same structure as origin)
    </ResponseField>
    
    <ResponseField name="scheduledOut" type="string">
      Scheduled departure time (ISO 8601)
    </ResponseField>
    
    <ResponseField name="scheduledOff" type="string">
      Scheduled takeoff time (ISO 8601)
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Flight status description
    </ResponseField>
    
    <ResponseField name="airline" type="object">
      Airline information

        <ResponseField name="name" type="string">

        <ResponseField name="iata" type="string">
          IATA code
        </ResponseField>
        
        <ResponseField name="icao" type="string">
          ICAO code
        </ResponseField>
        
        <ResponseField name="image" type="string">
          Airline logo URL
        </ResponseField>

    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="count" type="number">
  Number of flights found
</ResponseField>

## Get flight details

<CodeGroup>
```bash cURL
curl -X GET "https://api.aero.com/v1/flight?iata=UA123&icao=UAL123&timezone=America/Los_Angeles&faFlightId=UAL123-1710936000-airline-0123" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.aero.com/v1/flight?iata=UA123&icao=UAL123&timezone=America/Los_Angeles&faFlightId=UAL123-1710936000-airline-0123', {
  headers: {
    'Authorization': 'Bearer YOUR_JWT_TOKEN'
  }
});
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.aero.com/v1/flight',
    params={
        'iata': 'UA123',
        'icao': 'UAL123',
        'timezone': 'America/Los_Angeles',
        'faFlightId': 'UAL123-1710936000-airline-0123'
    },
    headers={'Authorization': 'Bearer YOUR_JWT_TOKEN'}
)
data = response.json()
```
</CodeGroup>

<ParamField path="GET" type="/v1/flight">
  Get detailed information about a specific flight including real-time tracking data. Use faFlightId to fetch a specific flight after searching.
</ParamField>

### Query parameters

<ParamField query="iata" type="string" required>
  IATA flight number
  
  **Example:** `UA123`
</ParamField>

<ParamField query="icao" type="string" required>
  ICAO flight number
  
  **Example:** `UAL123`
</ParamField>

<ParamField query="date" type="string">
  Flight date (ISO format). Defaults to current date if not provided
  
  **Example:** `2024-03-20`
</ParamField>

<ParamField query="timezone" type="string" required>
  The timezone from which the request is being made
  
  **Example:** `America/Los_Angeles`
</ParamField>

<ParamField query="forceUpdate" type="boolean" default="false">
  Force update the flight data (used to refresh the flight data)
</ParamField>

<ParamField query="faFlightId" type="string">
  FlightAware flight ID to fetch a specific flight. Use this after searching for flights to get details of a selected flight.
</ParamField>

## Get flight track/path

<CodeGroup>
```bash cURL
curl -X GET "https://api.aero.com/v1/flight/track?iata=UA123&icao=UAL123&date=2024-03-20&timezone=America/Los_Angeles" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.aero.com/v1/flight/track?iata=UA123&icao=UAL123&date=2024-03-20&timezone=America/Los_Angeles', {
  headers: {
    'Authorization': 'Bearer YOUR_JWT_TOKEN'
  }
});
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.aero.com/v1/flight/track',
    params={
        'iata': 'UA123',
        'icao': 'UAL123',
        'date': '2024-03-20',
        'timezone': 'America/Los_Angeles'
    },
    headers={'Authorization': 'Bearer YOUR_JWT_TOKEN'}
)
data = response.json()
```
</CodeGroup>

<ParamField path="GET" type="/v1/flight/track">
  Get the flight path with all position data points for a specific flight
</ParamField>

### Query parameters

<ParamField query="iata" type="string" required>
  IATA flight number
  
  **Example:** `UA123`
</ParamField>

<ParamField query="icao" type="string" required>
  ICAO flight number
  
  **Example:** `UAL123`
</ParamField>

<ParamField query="date" type="string">
  Flight date (ISO format). Defaults to current date if not provided
  
  **Example:** `2024-03-20`
</ParamField>

<ParamField query="timezone" type="string" required>
  The timezone from which the request is being made
  
  **Example:** `America/Los_Angeles` or `IST`
</ParamField>

### Response

<ResponseField name="actual_distance" type="number">
  Distance flown in miles as of the latest position point
  
  **Example:** `2450`
</ResponseField>

<ResponseField name="positions" type="array">
  Array of flight positions
  
  <Expandable title="Position object">
    <ResponseField name="fa_flight_id" type="string">
      FlightAware flight ID for this position
    </ResponseField>
    
    <ResponseField name="altitude" type="number">
      Aircraft altitude in hundreds of feet
      
      **Example:** `350` (35,000 feet)
    </ResponseField>
    
    <ResponseField name="altitude_change" type="string">
      Altitude change indicator: `C` (climbing), `D` (descending), `-` (level)
    </ResponseField>
    
    <ResponseField name="groundspeed" type="number">
      Groundspeed in knots
      
      **Example:** `450`
    </ResponseField>
    
    <ResponseField name="heading" type="number">
      Aircraft heading in degrees (0-360)
      
      **Example:** `270`
    </ResponseField>
    
    <ResponseField name="latitude" type="number">
      Latitude position
      
      **Example:** `40.7128`
    </ResponseField>
    
    <ResponseField name="longitude" type="number">
      Longitude position
      
      **Example:** `-74.0060`
    </ResponseField>
    
    <ResponseField name="timestamp" type="string">
      Time that position was received
      
      **Example:** `2024-03-20T14:30:00Z`
    </ResponseField>
    
    <ResponseField name="update_type" type="string">
      Update type: `P` (projected), `O` (oceanic), `Z` (radar), `A` (ADS-B), `M` (multilateration), `D` (datalink), `X` (surface), `S` (space-based), `V` (virtual)
    </ResponseField>
  </Expandable>
</ResponseField>
